// 测试移动端切换
var target = document.getElementsByClassName('m-head-nav')[0]
var index = 0
target.onclick = function () {
    var node = document.getElementsByClassName('m-head-nav-bd')[0]
    if(index%2 == 0){
        node.setAttribute('class','m-head-nav-bd m-head-nav-bd--on')
    }else{
        node.setAttribute('class','m-head-nav-bd')
    }
    index++
}
